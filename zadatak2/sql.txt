CREATE TABLE `praksa`.`Praktikanti` (
  `idPraktikant` INT NOT NULL AUTO_INCREMENT,
  `Ime` VARCHAR(45) NOT NULL,
  `Prezime` VARCHAR(45) NULL,
  PRIMARY KEY (`idPraktikant`),
  UNIQUE INDEX `idPraktikant_UNIQUE` (`idPraktikant` ASC));


  CREATE TABLE `praksa`.`Mentori` (
  `idMentor` INT NOT NULL AUTO_INCREMENT,
  `Ime` VARCHAR(45) NOT NULL,
  `Prezime` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`idMentor`),
  UNIQUE INDEX `idMentor_UNIQUE` (`idMentor` ASC));
